########################################################################
# Set up library
########################################################################
# List any C++ sources here. If there are no sources (e.g., because there
# is no block controller), then this directory will be skipped.
list(APPEND rfnoc_specinvert_sources
    specinvert_block_control.cpp
    specinvert_block_control.cpp
    #my_block_control.cpp
)
if(NOT rfnoc_specinvert_sources)
    MESSAGE(STATUS "No C++ sources... skipping lib/")
    return()
endif()

########################################################################
# Setup the include and linker paths
########################################################################
include_directories(
    ${CMAKE_SOURCE_DIR}/lib
    ${CMAKE_SOURCE_DIR}/include
    ${CMAKE_BINARY_DIR}/lib
    ${CMAKE_BINARY_DIR}/include
    ${UHD_INCLUDE_DIRS}
    ${Boost_INCLUDE_DIR}
)

link_directories(
    ${Boost_LIBRARY_DIRS}
)

add_library(rfnoc-specinvert SHARED
    ${rfnoc_specinvert_sources}
)
target_link_libraries(rfnoc-specinvert
    ${UHD_LIBRARIES}
    ${Boost_LIBRARIES}
)
set_target_properties(rfnoc-specinvert
    PROPERTIES DEFINE_SYMBOL "RFNOC_SPECINVERT_EXPORTS")

file(GENERATE OUTPUT ${CMAKE_BINARY_DIR}/lib/rfnoc-specinvert CONTENT "$<TARGET_FILE_NAME:rfnoc-specinvert>\n")

########################################################################
# Install built library files
########################################################################
install(TARGETS rfnoc-specinvert
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR} # .so/.dylib file
    ARCHIVE DESTINATION ${CMAKE_INSTALL_LIBDIR} # .lib file
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR} # .dll file
)
install(FILES
    ${CMAKE_BINARY_DIR}/lib/rfnoc-specinvert
    DESTINATION ${MODULES_D_DIR}
    COMPONENT libraries
)
